def print_board(board):
  for row in board:
      print(" | ".join(row))
      print("-" * 9)

def initialize_board():
    return [[" " for _ in range(3)] for _ in range(3)]

def main():
    board = initialize_board()
    print_board(board)

if __name__ == "__main__":
    main()
